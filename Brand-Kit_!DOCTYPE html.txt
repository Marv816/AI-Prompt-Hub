<script>
let userProfile = { company: "Your Company", niche: "Business", logo: null };

const DOCUMENTS = [
  { name: "Logo Identity Prompt", icon: "🎨", type: "AI Tool" },
  { name: "Agreement for Reseller", icon: "🤝", type: "Legal" },
  { name: "Consultant Agreement", icon: "📋", type: "Legal" },
  { name: "Asset Purchase", icon: "🏛️", type: "Finance" },
  { name: "Sponsorship Agreement", icon: "✨", type: "Marketing" }
];

function buildGrid() {
  document.getElementById('templateGrid').innerHTML = DOCUMENTS.map(doc => `
    <div class="file-card" onclick="openDoc('${doc.name}')">
      <div style="font-size: 3rem; margin-bottom: 1rem;">${doc.icon}</div>
      <div class="file-name">${doc.name}</div>
      <div style="color: var(--accent3); font-size: 0.7rem; font-weight: 800; text-transform: uppercase;">${doc.type}</div>
    </div>
  `).join('');
}

function openDoc(name) {
    document.getElementById('editTitle').innerText = name;
    const fields = document.getElementById('dynamicFields');
    
    if(name === "Logo Identity Prompt") {
        fields.innerHTML = `
            <div style="background: var(--bg); padding: 1.5rem; border-radius: 10px; border: 1px solid var(--accent); margin-top: 1rem;">
                <label>Architected Logo Prompt:</label>
                <textarea style="height: 150px; margin-top: 10px;" id="logoPrompt">Act as a professional Brand Designer. Create a minimalist logo for "${userProfile.company}", a company in the ${userProfile.niche} industry. The design should be clean, high-contrast, and suitable for both web and print. Use a professional color palette of deep violets and cyan.</textarea>
                <button class="btn-accent" style="margin-top: 10px;" onclick="copyLogoPrompt()">📋 Copy Prompt</button>
            </div>
        `;
    } else {
        fields.innerHTML = `
            <div class="field-group"><label>Client Name</label><input type="text"></div>
            <div class="field-group"><label>Effective Date</label><input type="date"></div>
        `;
    }
    document.getElementById('editorOverlay').style.display = 'flex';
}

function handleLogo(e) {
    const reader = new FileReader();
    reader.onload = function() {
        userProfile.logo = reader.result;
        document.getElementById('logoDisplay').innerHTML = `<img src="${reader.result}">`;
    }
    reader.readAsDataURL(e.target.files[0]);
}

function openProfile() { document.getElementById('profileModal').style.display = 'flex'; }

function saveProfile() {
    userProfile.company = document.getElementById('prof-company').value;
    userProfile.niche = document.getElementById('prof-niche').value;
    document.getElementById('displayCompany').innerText = userProfile.company;
    localStorage.setItem('officeProfile', JSON.stringify(userProfile));
    document.getElementById('profileModal').style.display = 'none';
}

function copyLogoPrompt() {
    const txt = document.getElementById('logoPrompt').value;
    navigator.clipboard.writeText(txt);
    alert("Logo Prompt Copied!");
}

function showView(view) {
    document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
    document.getElementById('view-' + view).classList.add('active');
}

window.onload = () => {
    buildGrid();
    const saved = localStorage.getItem('officeProfile');
    if(saved) {
        userProfile = JSON.parse(saved);
        document.getElementById('displayCompany').innerText = userProfile.company;
        if(userProfile.logo) document.getElementById('logoDisplay').innerHTML = `<img src="${userProfile.logo}">`;
    }
};
</script>

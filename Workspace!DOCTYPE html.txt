<script>
let workspace = [];

const DOCUMENTS = [
  { name: "Agreement for Reseller", icon: "🤝", type: "Legal" },
  { name: "Consultant Agreement", icon: "📋", type: "Legal" },
  { name: "Asset Purchase Agreement", icon: "🏛️", type: "Finance" },
  { name: "Sponsorship Agreement", icon: "✨", type: "Marketing" }
];

function buildGrid() {
  document.getElementById('templateGrid').innerHTML = DOCUMENTS.map(doc => `
    <div class="file-card" onclick="openDoc('${doc.name}')">
      <div style="font-size: 3rem; margin-bottom: 1rem;">${doc.icon}</div>
      <div class="file-name">${doc.name}</div>
      <div style="color: var(--accent3); font-size: 0.7rem; font-weight: 800;">${doc.type}</div>
    </div>
  `).join('');
}

function openDoc(name) {
    document.getElementById('editTitle').innerText = name;
    document.getElementById('work-title').value = name + " Draft";
    document.getElementById('work-content').value = `Agreement for: ${name}\nDate: ${new Date().toLocaleDateString()}\n\n[Your content will be generated here]`;
    document.getElementById('editorOverlay').style.display = 'flex';
}

// ── SAVE TO WORKSPACE (LOCALSTORAGE) ──
function saveToWorkspace() {
    const item = {
        id: Date.now(),
        title: document.getElementById('work-title').value,
        content: document.getElementById('work-content').value,
        date: new Date().toLocaleDateString()
    };
    workspace.push(item);
    localStorage.setItem('userWorkspace', JSON.stringify(workspace));
    renderWorkspace();
    alert("Saved to your personal workspace!");
}

function renderWorkspace() {
    const list = document.getElementById('workspaceList');
    if (workspace.length === 0) {
        list.innerHTML = `<p style="text-align:center; padding: 3rem; border: 1px dashed var(--border); border-radius: 12px;">No saved items yet.</p>`;
        return;
    }
    list.innerHTML = workspace.map(item => `
        <div class="workspace-item">
            <div>
                <div style="font-weight:700;">${item.title}</div>
                <div style="font-size:0.8rem; color:var(--text-muted)">Saved on ${item.date}</div>
            </div>
            <div style="display:flex; gap:10px;">
                <button class="btn-outline" style="padding: 0.4rem 0.8rem;" onclick="loadFromWorkspace(${item.id})">Open</button>
                <button class="btn-outline" style="padding: 0.4rem 0.8rem; color: #ff4d4d;" onclick="deleteFromWorkspace(${item.id})">Delete</button>
            </div>
        </div>
    `).join('');
}

// ── DOWNLOAD FUNCTION (GENERATES REAL FILE) ──
function downloadDoc() {
    const title = document.getElementById('work-title').value;
    const text = document.getElementById('work-content').value;
    const blob = new Blob([text], { type: 'application/msword' });
    const link = document.createElement('a');
    link.href = URL.createObjectURL(blob);
    link.download = title + ".doc";
    link.click();
}

function loadFromWorkspace(id) {
    const item = workspace.find(i => i.id === id);
    document.getElementById('editTitle').innerText = "Edit Saved Work";
    document.getElementById('work-title').value = item.title;
    document.getElementById('work-content').value = item.content;
    document.getElementById('editorOverlay').style.display = 'flex';
}

function deleteFromWorkspace(id) {
    workspace = workspace.filter(i => i.id !== id);
    localStorage.setItem('userWorkspace', JSON.stringify(workspace));
    renderWorkspace();
}

function showView(view) {
    document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
    document.querySelectorAll('.btn-outline').forEach(b => b.classList.remove('btn-accent'));
    document.getElementById('view-' + view).classList.add('active');
}

function closeEditor() { document.getElementById('editorOverlay').style.display = 'none'; }

window.onload = () => {
    buildGrid();
    const saved = localStorage.getItem('userWorkspace');
    if(saved) workspace = JSON.parse(saved);
    renderWorkspace();
};
</script>
